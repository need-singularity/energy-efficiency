tests = [

    status = "PASS" if val == formula else "FAIL"

print(f"\nEXACT: {exact}/{total} ({100*exact//total}%)")
assert exact == total, f"FAIL: {exact}/{total}"
print("전체 PASS — 법학/사법 n=6 천장 확인")
