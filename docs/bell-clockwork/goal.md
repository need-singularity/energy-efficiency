## 검증코드

```python
"""n=6 종/시계 기계학 검증 -- 하드코딩 금지, n에서 도출"""
import math

def sigma(n):
    return sum(d for d in range(1, n+1) if n % d == 0)

def tau(n):
    return sum(1 for d in range(1, n+1) if n % d == 0)

def phi(n):
    return sum(1 for k in range(1, n+1) if math.gcd(k, n) == 1)

def J2(n):
    return sum(k*k for k in range(1, n+1) if math.gcd(k, n) == 1)

def sopfr(n):
    s, d = 0, 2
    tmp = n
    while d * d <= tmp:
        while tmp % d == 0:
            s += d
            tmp //= d
        d += 1
    if tmp > 1:
        s += tmp
    return s

n = 6
s, t, p, j2, sp = sigma(n), tau(n), phi(n), J2(n), sopfr(n)

# BC-01: 시계 12시간 = sigma(6)
assert s == 12, f"sigma({n})={s}, 12 예상"
print(f"BC-01 PASS: sigma({n}) = {s} = 12시간")

# BC-02: 하루 24시간 = J2(6)
assert j2 == 24, f"J2({n})={j2}, 24 예상"
print(f"BC-02 PASS: J2({n}) = {j2} = 24시간")

# BC-03: 60분 = n * (n + tau)
assert n * (n + t) == 60, f"{n}*({n}+{t})={n*(n+t)}, 60 예상"
print(f"BC-03 PASS: {n} * ({n}+{t}) = {n*(n+t)} = 60분")

# BC-04: 탈진기 2상 = phi(6)
assert p == 2, f"phi({n})={p}, 2 예상"
print(f"BC-04 PASS: phi({n}) = {p} = 탈진기 2상")

# BC-05: 종 배음비 = n의 약수 {1,2,3,6}
divisors = [d for d in range(1, n+1) if n % d == 0]
assert divisors == [1, 2, 3, 6], f"약수={divisors}"
print(f"BC-05 PASS: {n}의 약수 = {divisors} (배음비 1:2:3:6)")

# BC-06: 기어 최적 톱니수 = sigma 배수
for teeth in [12, 24, 36, 48, 60]:
    assert teeth % s == 0, f"{teeth} % {s} != 0"
print(f"BC-06 PASS: 표준 톱니수 12,24,36,48,60 모두 sigma={s}의 배수")

# n=5 대조
n5 = 5
s5, t5, p5, j25 = sigma(n5), tau(n5), phi(n5), J2(n5)
print(f"\n--- n=5 대조 ---")
print(f"sigma(5)={s5} (12시 불가), J2(5)={j25} (24시 불가)")
print(f"tau(5)={t5}, phi(5)={p5}")
assert s5 != 12, "n=5도 sigma=12이면 유일성 실패"
assert j25 != 24, "n=5도 J2=24이면 유일성 실패"
print("n=5 대조 PASS: n=6만 시계 구조와 정합")
```
